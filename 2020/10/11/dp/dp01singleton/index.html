    单例说明：

    • 单例模式类只能有一个实例

    • 单例模式类必须自己创建自己的唯一实例

    • 单例模式类必须给所有其他对象提供这一实例

      单例模式的特点：

    • 静态化实例

    • 私有化构造函数

    • 提供公共的获取方法

    单例的7种写法：

    1. 饿汉式

      /**
 * @author LMZ
 * 饿汉式加载
 * 优点：类加载到内存后，就实例化一个单例，JVM保证线程安全
 */
public class Pl1 {
    private static final Pl1 INSTANCE = new Pl1();

    private Pl1() {}

    public static Pl1 getInstance() {
        return INSTANCE;
    }

    public void t(){
        System.out.println("pl1");
    }

    public static void main(String[] args) {
        System.out.println(Pl1.getInstance().hashCode());
        System.out.println(Pl1.getInstance().hashCode());
    }
}

    2. 懒汉式

      /**
 * @author LMZ
 * 懒汉式
 * 线程不安全，多线程访问时出现返回实例不唯一
 */
public class Pl2 {
    private static Pl2 INSTANCE;

    private Pl2() {}

    public static Pl2 getInstance(){
        if (INSTANCE == null) {
            try {
                Thread.sleep(1);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
            INSTANCE = new Pl2();
        }
        return INSTANCE;
    }

    public void t(){
        System.out.println("pl2");
    }

    public static void main(String[] args) {
        for (int i = 0; i < 100; i++) {
            new Thread(()->{
                System.out.println(Pl2.getInstance().hashCode());
            }).start();
        }
    }
}

    3. 懒汉式，加锁

      /**
 * @author LMZ
 * 懒汉式
 * 在getInstance方法上，加Synchronized锁保证线程安全，效率低
 */
public class Pl3 {
    private static Pl3 INSTANCE;

    private Pl3() {}

    public synchronized static Pl3 getInstance(){

    4. 线程不安全

      /**
 * @author LMZ
 * 线程不安全；
 * 在INSTANCE判断后，多线程在获取锁可能出现锁等待
 */
public class Pl4 {
    private static Pl4 INSTANCE;

    private Pl4() {
    }

    public static Pl4 getInstance() {
        if (INSTANCE == null) {
            synchronized (Pl4.class) {
                try {
                    Thread.sleep(1);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
                INSTANCE = new Pl4();
            }
        }
        return INSTANCE;
    }

    public void t() {
        System.out.println("pl4");
    }

    public static void main(String[] args) {
        for (int i = 0; i < 100; i++) {
            new Thread(() -> {
                System.out.println(Pl4.getInstance().hashCode());
            }).start();
        }
    }
}

    5. 双重锁

      /**
 * @author LMZ
 * 双重锁检查
 */
public class Pl5 {
    private static Pl5 INSTANCE;

    private Pl5() {
    }

    public static Pl5 getInstance() {
        if (INSTANCE == null) {
            synchronized (Pl5.class) {
                if (INSTANCE == null) {
                    try {
                        Thread.sleep(1);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                    INSTANCE = new Pl5();
                }
            }
        }
        return INSTANCE;
    }

    public void t() {
        System.out.println("pl5");

}

    6. 静态内部类

      /**
 * @author LMZ
 * 静态内部类实现单例
 */
public class Pl6 {
    private Pl6(){}

    private static class Pl6Helper{
        private static Pl6 INSTANCE = new Pl6();
    }

    private static Pl6 getInstance() {
        return Pl6Helper.INSTANCE;
    }

    public void t() {
        System.out.println("pl6");
    }

    public static void main(String[] args) {
        for (int i = 0; i < 100; i++) {
            new Thread(() -> {
                System.out.println(Pl6.getInstance().hashCode());
            }).start();
        }
    }
}

    7. 枚举实现单例

      /**
 * @author LMZ
 * 枚举类实现单例
 * 解决线程同步与反序列化
 */
public enum Pl7 {
    INSTANCE;

    public static void main(String[] args) {
        for (int i = 0; i < 100; i++) {
            new Thread(() -> {
                System.out.println(Pl7.INSTANCE.hashCode());
            }).start();
        }
    }
}
