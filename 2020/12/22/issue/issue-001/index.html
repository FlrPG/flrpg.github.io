    线程池创建-中止

    问题集 issue
    发布时间 : 2020-12-22 18:08

    评论:

    依赖

    示例

    private static void test5ThreadDown() {
        int polGroupCount = 29999;
        int polPersonCount = 51000;
        int gSize = 8, pSize = 2;
        int gCount = polGroupCount / gSize, pCount = polPersonCount / pSize;
        //创建线程池
        ExecutorService service = ThreadUtil.newExecutor(gSize + pSize);
        for (int i = 0; i < gSize; i++) {
            int startIndex = i * gCount;
            int endIndex = i * gCount + gCount;
            if (i == gSize - 1) {
                endIndex = polGroupCount;
            }
            if (startIndex == endIndex) {
                continue;
            }
            int finalEndIndex = endIndex;
            service.submit(() -> {
                try {
                    System.out.println("startIndex: " + startIndex + " endIndex: " + finalEndIndex);
                    for (int j = 0; j < 5; j++) {
                        ThreadUtil.safeSleep(1000);
                        System.out.println(Thread.currentThread().getName() + "--j--" + j);
                    }
                    System.out.println(Thread.currentThread().getName() + "-end......");
                } catch (Exception e) {
                    System.out.println(e.getMessage());
                }
            });
        }
        //线程池
        service.shutdown();
        while (true) {
            if (service.isTerminated()) {
                //当所有任务执行结束完成后，return true
                System.out.println("线程都结束勒。。。");
                break;
            }
            ThreadUtil.safeSleep(1000);
        }
        System.out.println("endAll==================");
    }
